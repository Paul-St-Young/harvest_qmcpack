import numpy as np


[docs]def ft_iso3d(myk, myr, frm):
  fkm = [np.trapz(myr*np.sin(k*myr)/k*frm, myr) for k in myk]
  return 4*np.pi*np.array(fkm)



[docs]def gr2sk(myk, myr, grm, rho):
  skm = 1+rho*ft_iso3d(myk, myr, grm-1)
  return skm



[docs]def sk2gr(myr, myk, skm, rho):
  grm = 1+ft_iso3d(myr, myk, skm-1)/rho/(2*np.pi)**3
  return grm



[docs]def gofr_snapshot(rij, rcut, rho, nbin):
  natom = rij.shape[0]
  assert rij.shape[1] == natom
  idx = np.triu_indices(natom, k=1)
  dists = rij[idx]
  dr = rcut/(nbin+1)
  bin_edges = np.arange(nbin+1)*dr
  norm = np.diff(bin_edges**3)*(4./3*np.pi*rho)*(natom-1)/2
  counts, _ = np.histogram(dists, bin_edges)
  gr = counts/norm
  myr = 0.5*(bin_edges[:-1]+bin_edges[1:])
  return myr, gr
